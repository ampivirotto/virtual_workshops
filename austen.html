packages = ["pandas", "re"] import re sample_text = "Pride and Prejudice is a novel by Jane Austen. It is full of wit and social commentary." words = re.findall(r'\b\w+\b', sample_text.lower()) print("Words:", words) print("Total words:", len(words))

stop_words = {"and", "is", "a", "it", "of"} filtered_words = [w for w in words if w not in stop_words] print("Filtered words:", filtered_words)

Step 3: Visualizing Sentiment Trajectories

Here is a small interactive demo of sentiment trajectories using preloaded data. Try editing the code and re-running it!

packages = ["pandas", "matplotlib"] import pandas as pd import matplotlib.pyplot as plt # Dummy sentiment data data = { "Book": ["Pride and Prejudice"]*5 + ["Emma"]*5, "Segment": [1,2,3,4,5]*2, "Sentiment": [0.1,0.3,0.2,-0.1,0.0, 0.2,0.1,-0.2,0.0,0.1] } df = pd.DataFrame(data) fig, ax = plt.subplots(figsize=(6,4)) for book, subset in df.groupby("Book"): ax.plot(subset["Segment"], subset["Sentiment"], marker='o', label=book) ax.set_title("Sentiment Trajectories Across Austen Novels") ax.set_xlabel("Segment") ax.set_ylabel("Sentiment Score") ax.axhline(0, color='gray', linestyle='--') ax.legend() plt.show()
